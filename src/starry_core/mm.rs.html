13/// Creates a new empty user address space.
14pub fn new_user_aspace_empty() -> AxResult<AddrSpace> {
15    AddrSpace::new_empty(
16        VirtAddr::from_usize(axconfig::plat::USER_SPACE_BASE),
17        axconfig::plat::USER_SPACE_SIZE,
18    )
19}

21/// If the target architecture requires it, the kernel portion of the address
22/// space will be copied to the user address space.
23pub fn copy_from_kernel(aspace: &mut AddrSpace) -> AxResult {
24    if !cfg!(target_arch = "aarch64") && !cfg!(target_arch = "loongarch64") {
25        // ARMv8 (aarch64) and LoongArch64 use separate page tables for user space
26        // (aarch64: TTBR0_EL1, LoongArch64: PGDL), so there is no need to copy the
27        // kernel portion to the user page table.
28        aspace.copy_mappings_from(&kernel_aspace().lock())?;
29    }
30    Ok(())
31}

33/// Map the signal trampoline to the user address space.
34pub fn map_trampoline(aspace: &mut AddrSpace) -> AxResult {
35    let signal_trampoline_paddr = virt_to_phys(axsignal::arch::signal_trampoline_address().into());
36    aspace.map_linear(
37        axconfig::plat::SIGNAL_TRAMPOLINE.into(),
38        signal_trampoline_paddr,
39        PAGE_SIZE_4K,
40        MappingFlags::READ | MappingFlags::EXECUTE | MappingFlags::USER,
41    )?;
42    Ok(())
43}

45/// Map the elf file to the user address space.
46///
47/// # Arguments
48/// - `uspace`: The address space of the user app.
49/// - `elf`: The elf file.
50///
51/// # Returns
52/// - The entry point of the user app.
53fn map_elf(uspace: &mut AddrSpace, elf: &ElfFile) -> AxResult<(VirtAddr, [AuxvEntry; 16])> {
54    let uspace_base = uspace.base().as_usize();
55    let elf_parser = ELFParser::new(
56        elf,
57        axconfig::plat::USER_INTERP_BASE,
58        Some(uspace_base as isize),
59        uspace_base,
60    )
61    .map_err(|_| AxError::InvalidData)?;
62
63    for segement in elf_parser.ph_load() {
64        debug!(
65            "Mapping ELF segment: [{:#x?}, {:#x?}) flags: {:#x?}",
66            segement.vaddr,
67            segement.vaddr + segement.memsz as usize,
68            segement.flags
69        );
70        let seg_pad = segement.vaddr.align_offset_4k();
71        assert_eq!(seg_pad, segement.offset % PAGE_SIZE_4K);
72
73        let seg_align_size =
74            (segement.memsz as usize + seg_pad + PAGE_SIZE_4K - 1) & !(PAGE_SIZE_4K - 1);
75        uspace.map_alloc(
76            segement.vaddr.align_down_4k(),
77            seg_align_size,
78            segement.flags,
79            true,
80        )?;
81        let seg_data = elf
82            .input
83            .get(segement.offset..segement.offset + segement.filesz as usize)
84            .ok_or(AxError::InvalidData)?;
85        uspace.write(segement.vaddr, seg_data)?;
86        // TDOO: flush the I-cache
87    }
88
89    Ok((
90        elf_parser.entry().into(),
91        elf_parser.auxv_vector(PAGE_SIZE_4K),
92    ))
93}

95/// Load the user app to the user address space.
96///
97/// # Arguments
98/// - `uspace`: The address space of the user app.
99/// - `args`: The arguments of the user app. The first argument is the path of the user app.
100/// - `envs`: The environment variables of the user app.
101///
102/// # Returns
103/// - The entry point of the user app.
104/// - The stack pointer of the user app.
105pub fn load_user_app(
106    uspace: &mut AddrSpace,
107    args: &[String],
108    envs: &[String],
109) -> AxResult<(VirtAddr, VirtAddr)> {
110    if args.is_empty() {
111        return Err(AxError::InvalidInput);
112    }
113    let file_data = axfs::api::read(args[0].as_str())?;
114    if file_data.starts_with(b"#!") {
115        let head = &file_data[2..file_data.len().min(256)];
116        let pos = head.iter().position(|c| *c == b'\n').unwrap_or(head.len());
117        let line = core::str::from_utf8(&head[..pos]).map_err(|_| AxError::InvalidData)?;
118
119        let new_args: Vec<String> = line
120            .splitn(2, |c: char| c.is_ascii_whitespace())
121            .map(|s| s.trim_ascii().to_owned())
122            .chain(args.iter().cloned())
123            .collect();
124        return load_user_app(uspace, &new_args, envs);
125    }
126    let elf = ElfFile::new(&file_data).map_err(|_| AxError::InvalidData)?;
127
128    if let Some(interp) = elf
129        .program_iter()
130        .find(|ph| ph.get_type() == Ok(xmas_elf::program::Type::Interp))
131    {
132        let interp = match interp.get_data(&elf) {
133            Ok(SegmentData::Undefined(data)) => data,
134            _ => panic!("Invalid data in Interp Elf Program Header"),
135        };
136
137        let mut interp_path = axfs::api::canonicalize(
138            CStr::from_bytes_with_nul(interp)
139                .map_err(|_| AxError::InvalidData)?
140                .to_str()
141                .map_err(|_| AxError::InvalidData)?,
142        )?;
143
144        if interp_path == "/lib/ld-linux-riscv64-lp64.so.1"
145            || interp_path == "/lib64/ld-linux-loongarch-lp64d.so.1"
146            || interp_path == "/lib64/ld-linux-x86-64.so.2"
147            || interp_path == "/lib/ld-linux-aarch64.so.1"
148        {
149            // TODO: Use soft link
150            interp_path = String::from("/musl/lib/libc.so");
151        }
152
153        // Set the first argument to the path of the user app.
154        let mut new_args = vec![interp_path];
155        new_args.extend_from_slice(args);
156        return load_user_app(uspace, &new_args, envs);
157    }
158
159    let (entry, mut auxv) = map_elf(uspace, &elf)?;
160    // The user stack is divided into two parts:
161    // `ustack_start` -> `ustack_pointer`: It is the stack space that users actually read and write.
162    // `ustack_pointer` -> `ustack_end`: It is the space that contains the arguments, environment variables and auxv passed to the app.
163    //  When the app starts running, the stack pointer points to `ustack_pointer`.
164    let ustack_end = VirtAddr::from_usize(axconfig::plat::USER_STACK_TOP);
165    let ustack_size = axconfig::plat::USER_STACK_SIZE;
166    let ustack_start = ustack_end - ustack_size;
167    debug!(
168        "Mapping user stack: {:#x?} -> {:#x?}",
169        ustack_start, ustack_end
170    );
171
172    let stack_data = app_stack_region(args, envs, &mut auxv, ustack_start, ustack_size);
173    uspace.map_alloc(
174        ustack_start,
175        ustack_size,
176        MappingFlags::READ | MappingFlags::WRITE | MappingFlags::USER,
177        true,
178    )?;
179
180    let heap_start = VirtAddr::from_usize(axconfig::plat::USER_HEAP_BASE);
181    let heap_size = axconfig::plat::USER_HEAP_SIZE;
182    uspace.map_alloc(
183        heap_start,
184        heap_size,
185        MappingFlags::READ | MappingFlags::WRITE | MappingFlags::USER,
186        true,
187    )?;
188
189    let user_sp = ustack_end - stack_data.len();
190
191    uspace.write(user_sp, stack_data.as_slice())?;
192
193    Ok((entry, user_sp))
194}

196#[percpu::def_percpu]
197static mut ACCESSING_USER_MEM: bool = false;
198
199/// Enables scoped access into user memory, allowing page faults to occur inside
200/// kernel.
201pub fn access_user_memory<R>(f: impl FnOnce() -> R) -> R {
202    ACCESSING_USER_MEM.with_current(|v| {
203        *v = true;
204        let result = f();
205        *v = false;
206        result
207    })
208}
209
210/// Check if the current thread is accessing user memory.
211pub fn is_accessing_user_memory() -> bool {
212    ACCESSING_USER_MEM.read_current()
