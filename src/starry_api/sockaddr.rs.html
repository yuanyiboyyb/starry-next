10use axerrno::{LinuxError, LinuxResult};
11use linux_raw_sys::net::{
12    __kernel_sa_family_t, AF_INET, AF_INET6, in_addr, in6_addr, sockaddr, sockaddr_in,
13    sockaddr_in6, socklen_t,

16/// A type that can hold any kind of socket address, as a safe abstraction for
17/// `sockaddr`.
18///
19/// Socket addresses can be converted to `SocketAddrAny` via the [`From`] and
20/// [`Into`] traits. `SocketAddrAny` can be converted back to a specific socket
21/// address type with [`TryFrom`] and [`TryInto`]. These implementations return
22/// [`LinuxError::EAFNOSUPPORT`] if the address family does not match the requested
23/// type.
24#[derive(Clone)]
25pub struct SockAddr {
26    // Invariants:
27    //  - `len` is at least `size_of::<__kernel_sa_family_t>()`
28    //  - `len` is at most `size_of::<sockaddr>()`
29    //  - The first `len` bytes of `storage` are initialized.
30    len: socklen_t,
31    storage: MaybeUninit<sockaddr>,
32}
33
34// SAFETY: Bindgen adds a union with a raw pointer for alignment but it's never
35// used. `sockaddr` is just a bunch of bytes and it doesn't hold pointers.
36unsafe impl Send for SockAddr {}
37
38// SAFETY: Same as with `Send`.
39unsafe impl Sync for SockAddr {}
40
41impl SockAddr {
42    /// Creates a socket address from reading from `ptr`, which points at `len`
43    /// initialized bytes.
44    ///
45    /// Returns [`LinuxError::EINVAL`] if `len` is smaller than `__kernel_sa_family_t` or larger than
46    /// `sockaddr`.
47    ///
48    /// # Safety
49    ///
50    ///  - `ptr` must be a pointer to memory containing a valid socket address.
51    ///  - `len` bytes must be initialized.
52    pub unsafe fn read(ptr: *const sockaddr, len: socklen_t) -> LinuxResult<Self> {
53        if size_of::<__kernel_sa_family_t>() < len as usize || len as usize > size_of::<sockaddr>()
54        {
55            return Err(LinuxError::EINVAL);
56        }
57        let mut storage = MaybeUninit::<sockaddr>::uninit();
58        unsafe {
59            core::ptr::copy_nonoverlapping(
60                ptr as *const u8,
61                storage.as_mut_ptr() as *mut u8,
62                len as usize,
63            )
64        };
65        Ok(Self { storage, len })
66    }
67
68    /// Gets the address family of this socket address.
69    #[inline]
70    pub fn family(&self) -> u32 {
71        // SAFETY: Our invariants maintain that the `sa_family` field is
72        // initialized.
73        unsafe {
74            self.storage
75                .assume_init_ref()
76                .__storage
77                .__bindgen_anon_1
78                .__bindgen_anon_1
79                .ss_family as u32
80        }
81    }
82
83    /// Returns the length of the encoded sockaddr.
84    #[inline]
85    pub fn addr_len(&self) -> socklen_t {
86        self.len
87    }
88
89    /// Gets the initialized part of the storage as bytes.
90    #[inline]
91    pub fn bytes(&self) -> &[u8] {
92        unsafe { core::slice::from_raw_parts(self.storage.as_ptr().cast(), self.len as usize) }
93    }
94}

96impl From<SocketAddrV4> for SockAddr {
97    fn from(v4: SocketAddrV4) -> Self {
98        let addr = sockaddr_in {
99            sin_family: AF_INET as _,
100            sin_port: v4.port().to_be(),
101            sin_addr: in_addr {
102                s_addr: u32::from_ne_bytes(v4.ip().octets()),
103            },
104            __pad: [0_u8; 8],
105        };
106        unsafe {
107            Self::read(
108                &addr as *const sockaddr_in as *const sockaddr,
109                core::mem::size_of::<sockaddr_in>() as socklen_t,
110            )
111            .unwrap()
112        }
113    }
114}

116impl From<SocketAddrV6> for SockAddr {
117    fn from(v6: SocketAddrV6) -> Self {
118        let addr = sockaddr_in6 {
119            sin6_family: AF_INET6 as _,
120            sin6_port: v6.port().to_be(),
121            sin6_flowinfo: v6.flowinfo().to_be(),
122            sin6_addr: in6_addr {
123                in6_u: linux_raw_sys::net::in6_addr__bindgen_ty_1 {
124                    u6_addr8: v6.ip().octets(),
125                },
126            },
127            sin6_scope_id: v6.scope_id(),
128        };
129        unsafe {
130            Self::read(
131                &addr as *const sockaddr_in6 as *const sockaddr,
132                core::mem::size_of::<sockaddr_in6>() as socklen_t,
133            )
134            .unwrap()
135        }
136    }
137}

139impl From<SocketAddr> for SockAddr {
140    fn from(addr: SocketAddr) -> Self {
141        match addr {
142            SocketAddr::V4(v4) => v4.into(),
143            SocketAddr::V6(v6) => v6.into(),
144        }
145    }
146}

148impl TryFrom<SockAddr> for SocketAddrV4 {
149    type Error = LinuxError;
150
151    fn try_from(addr: SockAddr) -> LinuxResult<Self> {
152        if addr.family() != AF_INET {
153            return Err(LinuxError::EAFNOSUPPORT);
154        }
155        if size_of::<sockaddr_in>() < addr.addr_len() as usize {
156            return Err(LinuxError::EINVAL);
157        }
158        let addr = unsafe { &*(addr.storage.as_ptr() as *const sockaddr_in) };
159        Ok(SocketAddrV4::new(
160            Ipv4Addr::from_bits(u32::from_be(addr.sin_addr.s_addr)),
161            u16::from_be(addr.sin_port),
162        ))
163    }
164}

166impl TryFrom<SockAddr> for SocketAddrV6 {
167    type Error = LinuxError;
168
169    fn try_from(addr: SockAddr) -> LinuxResult<Self> {
170        if addr.family() != AF_INET6 {
171            return Err(LinuxError::EAFNOSUPPORT);
172        }
173        if size_of::<sockaddr_in6>() < addr.addr_len() as usize {
174            return Err(LinuxError::EINVAL);
175        }
176        let addr = unsafe { &*(addr.storage.as_ptr() as *const sockaddr_in6) };
177        Ok(SocketAddrV6::new(
178            Ipv6Addr::from(unsafe { addr.sin6_addr.in6_u.u6_addr8 }),
179            u16::from_be(addr.sin6_port),
180            u32::from_be(addr.sin6_flowinfo),
181            addr.sin6_scope_id,
182        ))
183    }
184}

186impl TryFrom<SockAddr> for SocketAddr {
187    type Error = LinuxError;
188
189    fn try_from(addr: SockAddr) -> LinuxResult<Self> {
190        match addr.family() {
191            AF_INET => Ok(SocketAddr::V4(addr.try_into()?)),
192            AF_INET6 => Ok(SocketAddr::V6(addr.try_into()?)),
193            _ => Err(LinuxError::EAFNOSUPPORT),
194        }
195    }
